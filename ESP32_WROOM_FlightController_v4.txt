/*
=================================================
 CYBER HAWK v6.5 – STABLE FLIGHT CONTROLLER
 Angle + Gyro Safe Flip + Gyro Filter
=================================================
*/

#include <Wire.h>
#include <MPU6050.h>

MPU6050 mpu;

// ================= UART =================
#define RXD2 16
#define TXD2 17

// ================= Motors =================
#define M1 25
#define M2 26
#define M3 32
#define M4 33

// ================= PWM =================
const int PWM_FREQ = 20000;
const int PWM_RES  = 8;

// ================= PID =================
float Kp = 1.3;
float Ki = 0.04;
float Kd = 0.25;
float yawKp = 1.5;

// ================= IMU =================
int16_t ax, ay, az;
int16_t gx, gy, gz;

float roll = 0;
float pitch = 0;
float yawRate = 0;

// gyro filter
float gxFiltered=0;
float gyFiltered=0;
float gzFiltered=0;

float gyroXoff=0;
float gyroYoff=0;
float gyroZoff=0;

// ================= PID memory =================
float prevRollError = 0;
float prevPitchError = 0;

float rollIntegral = 0;
float pitchIntegral = 0;

float pidRoll = 0;
float pidPitch = 0;
float pidYaw = 0;

// ================= Control =================
float rollSet = 0;
float pitchSet = 0;
float yawSet = 0;

int baseThrottle = 0;

bool armed = false;
bool hoverMode = false;

// ================= Hover =================
float hoverThrottle = 130;
float throttleFiltered = 0;

// ================= Flip =================
bool flipActive = false;
bool flipFront = true;

float flipStartPitch = 0;
bool flipStarted = false;

// ================= Timing =================
unsigned long lastTime = 0;
unsigned long lastSignal = 0;
unsigned long lastPID = 0;

// =================================================

void setup() {

Serial.begin(115200);
Serial2.begin(115200, SERIAL_8N1, RXD2, TXD2);

Wire.begin();
mpu.initialize();

delay(1000);
calibrateMPU();

// PWM
ledcSetup(0,PWM_FREQ,PWM_RES);
ledcSetup(1,PWM_FREQ,PWM_RES);
ledcSetup(2,PWM_FREQ,PWM_RES);
ledcSetup(3,PWM_FREQ,PWM_RES);

ledcAttachPin(M1,0);
ledcAttachPin(M2,1);
ledcAttachPin(M3,2);
ledcAttachPin(M4,3);

lastTime = millis();
lastPID = millis();
}

// =================================================

void loop(){

readIMU();
readSerial();
computePID();
handleFlip();
mixMotors();
failsafe();

}

// =================================================
// MPU CALIBRATION
// =================================================

void calibrateMPU(){

for(int i=0;i<1000;i++){

mpu.getMotion6(&ax,&ay,&az,&gx,&gy,&gz);

gyroXoff+=gx;
gyroYoff+=gy;
gyroZoff+=gz;

delay(3);
}

gyroXoff/=1000;
gyroYoff/=1000;
gyroZoff/=1000;

}

// =================================================
// IMU READ + FILTER
// =================================================

void readIMU(){

mpu.getMotion6(&ax,&ay,&az,&gx,&gy,&gz);

gx -= gyroXoff;
gy -= gyroYoff;
gz -= gyroZoff;

// gyro low-pass filter
gxFiltered = gxFiltered*0.7 + gx*0.3;
gyFiltered = gyFiltered*0.7 + gy*0.3;
gzFiltered = gzFiltered*0.7 + gz*0.3;

float accelRoll  = atan2(ay,az) * 57.2958;
float accelPitch = atan2(-ax, sqrt(ay*ay + az*az)) * 57.2958;

float dt = (millis() - lastTime)/1000.0;
lastTime = millis();

roll  = 0.98*(roll  + gxFiltered/131.0 * dt) + 0.02*accelRoll;
pitch = 0.98*(pitch + gyFiltered/131.0 * dt) + 0.02*accelPitch;

yawRate = gzFiltered/131.0;

}

// =================================================
// PID
// =================================================

void computePID(){

float dt = (millis() - lastPID)/1000.0;
lastPID = millis();

dt = constrain(dt,0.002,0.02);

float rollError = rollSet - roll;
float pitchError = pitchSet - pitch;

rollIntegral += rollError * dt;
pitchIntegral += pitchError * dt;

rollIntegral = constrain(rollIntegral,-100,100);
pitchIntegral = constrain(pitchIntegral,-100,100);

float rollDerivative = (rollError - prevRollError)/dt;
float pitchDerivative = (pitchError - prevPitchError)/dt;

// TPA
float tpa = 1.0;

if(baseThrottle > 150)
tpa = 1.0 - ((baseThrottle-150)*0.003);

pidRoll  = tpa * ((Kp*rollError) + (Ki*rollIntegral) + (Kd*rollDerivative));
pidPitch = tpa * ((Kp*pitchError) + (Ki*pitchIntegral) + (Kd*pitchDerivative));

pidYaw = yawKp*(yawSet - yawRate);
pidYaw = constrain(pidYaw,-150,150);

prevRollError = rollError;
prevPitchError = pitchError;

}

// =================================================
// SAFE ANGLE + GYRO FLIP
// =================================================

void handleFlip(){

if(!flipActive) return;

if(!flipStarted){

flipStartPitch = pitch;
flipStarted = true;

}

// disable stabilization
pidRoll = 0;
pidYaw = 0;

// throttle boost
throttleFiltered = baseThrottle + 60;

// flip torque
if(flipFront)
pidPitch = -200;
else
pidPitch = 200;

// detect rotation + slowing gyro
if(abs(pitch - flipStartPitch) > 150 && abs(gyFiltered) < 120){

flipActive=false;
flipStarted=false;

rollIntegral=0;
pitchIntegral=0;

}

}

// =================================================
// MOTOR MIX
// =================================================

void mixMotors(){

if(!armed){

ledcWrite(0,0);
ledcWrite(1,0);
ledcWrite(2,0);
ledcWrite(3,0);
return;

}

throttleFiltered += (baseThrottle - throttleFiltered)*0.2;

int base = hoverMode ? hoverThrottle : throttleFiltered;

int m1 = base + pidRoll - pidPitch - pidYaw;
int m2 = base - pidRoll - pidPitch + pidYaw;
int m3 = base - pidRoll + pidPitch - pidYaw;
int m4 = base + pidRoll + pidPitch + pidYaw;

m1 = constrain(m1,0,255);
m2 = constrain(m2,0,255);
m3 = constrain(m3,0,255);
m4 = constrain(m4,0,255);

ledcWrite(0,m1);
ledcWrite(1,m2);
ledcWrite(2,m3);
ledcWrite(3,m4);

}

// =================================================
// SERIAL CONTROL
// =================================================

void readSerial(){

if(!Serial2.available()) return;

String cmd = Serial2.readStringUntil('\n');
cmd.trim();

lastSignal = millis();

if(cmd=="ARM") armed=true;

else if(cmd=="DISARM"){
armed=false;
baseThrottle=0;
}

else if(cmd.startsWith("THR:")){
baseThrottle = cmd.substring(4).toInt();
baseThrottle = constrain(baseThrottle,0,255);
}

else if(cmd=="HOVER"){
hoverMode = !hoverMode;
}

else if(cmd=="FLIP_FRONT"){
if(baseThrottle>140){
flipActive=true;
flipFront=true;
}
}

else if(cmd=="FLIP_BACK"){
if(baseThrottle>140){
flipActive=true;
flipFront=false;
}
}

}

// =================================================
// FAILSAFE
// =================================================

void failsafe(){

if(millis() - lastSignal > 1000){

armed=false;

ledcWrite(0,0);
ledcWrite(1,0);
ledcWrite(2,0);
ledcWrite(3,0);

}

}
