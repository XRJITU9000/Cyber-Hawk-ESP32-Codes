 Compatible with WROOM v6

#include "esp_camera.h"
#include <WiFi.h>
#include <WebServer.h>

#define RXD2 3
#define TXD2 1

// ==== WIFI ====
const char* ssid = "YOUR_WIFI_NAME";
const char* password = "YOUR_WIFI_PASSWORD";

WebServer server(80);

// ===== CAMERA STREAM HANDLER =====
void handleStream() {
  WiFiClient client = server.client();

  String response = 
    "HTTP/1.1 200 OK\r\n"
    "Content-Type: multipart/x-mixed-replace; boundary=frame\r\n\r\n";

  server.sendContent(response);

  while (client.connected()) {
    camera_fb_t * fb = esp_camera_fb_get();
    if (!fb) continue;

    server.sendContent("--frame\r\n");
    server.sendContent("Content-Type: image/jpeg\r\n\r\n");
    client.write(fb->buf, fb->len);
    server.sendContent("\r\n");

    esp_camera_fb_return(fb);

    if (!client.connected()) break;
  }
}

// ===== PHOTO CAPTURE =====
void handleCapture() {
  camera_fb_t * fb = esp_camera_fb_get();
  if (!fb) {
    server.send(500, "text/plain", "Camera Error");
    return;
  }

  server.sendHeader("Content-Type", "image/jpeg");
  server.sendHeader("Content-Disposition", "inline; filename=capture.jpg");
  server.send_P(200, "image/jpeg", (const char*)fb->buf, fb->len);

  esp_camera_fb_return(fb);
}

// ===== UART SEND FUNCTIONS =====
void sendJoystick(int throttle, int pitch, int roll, int yaw) {
  Serial2.print("J:");
  Serial2.print(throttle);
  Serial2.print(",");
  Serial2.print(pitch);
  Serial2.print(",");
  Serial2.print(roll);
  Serial2.print(",");
  Serial2.println(yaw);
}

void sendCommand(String cmd) {
  Serial2.print("CMD:");
  Serial2.println(cmd);
}

// ===== WEB HANDLERS =====
void handleJoystick() {
  int t = server.arg("t").toInt();
  int p = server.arg("p").toInt();
  int r = server.arg("r").toInt();
  int y = server.arg("y").toInt();

  sendJoystick(t, p, r, y);
  server.send(200, "text/plain", "OK");
}

void handleCommand() {
  String cmd = server.arg("cmd");
  sendCommand(cmd);
  server.send(200, "text/plain", "CMD SENT");
}

// ===== CAMERA INIT (OV3660) =====
void startCamera() {

  camera_config_t config;
  config.ledc_channel = LEDC_CHANNEL_0;
  config.ledc_timer = LEDC_TIMER_0;

  config.pin_d0 = 5;
  config.pin_d1 = 18;
  config.pin_d2 = 19;
  config.pin_d3 = 21;
  config.pin_d4 = 36;
  config.pin_d5 = 39;
  config.pin_d6 = 34;
  config.pin_d7 = 35;

  config.pin_xclk = 0;
  config.pin_pclk = 22;
  config.pin_vsync = 25;
  config.pin_href = 23;
  config.pin_sscb_sda = 26;
  config.pin_sscb_scl = 27;

  config.pin_pwdn = 32;
  config.pin_reset = -1;

  config.xclk_freq_hz = 20000000;
  config.pixel_format = PIXFORMAT_JPEG;

  // Best for drone latency
  config.frame_size = FRAMESIZE_QVGA;  // 320x240
  config.jpeg_quality = 12;            // 10–15 good balance
  config.fb_count = 2;

  esp_camera_init(&config);
}

void setup() {
  Serial.begin(115200);
  Serial2.begin(115200, SERIAL_8N1, RXD2, TXD2);

  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
  }

  startCamera();

  server.on("/stream", HTTP_GET, handleStream);
  server.on("/capture", HTTP_GET, handleCapture);
  server.on("/joystick", HTTP_GET, handleJoystick);
  server.on("/command", HTTP_GET, handleCommand);

  server.begin();
}

void loop() {
  server.handleClient();
}
